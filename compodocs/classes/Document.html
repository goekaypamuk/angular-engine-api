        for (const hc in this.hyperCubeList) {
            if (this.hyperCubeList.hasOwnProperty(hc)) {
                this.hyperCubeList[hc].getLayout();
            }

        for (const l in this.listList) {
            if (this.hyperCubeList.hasOwnProperty(l)) {
                this.listList[l].getLayout();
            }
        }
    }
